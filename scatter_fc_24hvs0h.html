def get_concordance(fc_df):
    sca_up = fc_df.sca_lfc > 0
    sca_down = fc_df.sca_lfc < 0
    intron_up = fc_df.intron_lfc > 0
    intron_down = fc_df.intron_lfc < 0
    
    return {
        'both_up': np.logical_and(sca_up, intron_up),
        'both_down': np.logical_and(sca_down, intron_down),
        'discordant_sca_up': np.logical_and(sca_up, intron_down),
        'discordant_sca_down': np.logical_and(sca_down, intron_up),
        'discordant': np.logical_or(
            np.logical_and(sca_up, intron_down),
            np.logical_and(sca_down, intron_up))
    }

def plot_sca_intron_corr(fc_df):
    
    sig_color = '#1c9099'
    
    interesting = fc_df.loc[fc_df.index.isin(genes_of_interest)]
    
    plt.figure(figsize=(9,9))
    plt.axhline(0, c='black')
    plt.axvline(0, c='black')
    both_sig = (fc_df.loc[:, ['intron_padj', 'sca_padj']] < 0.05).all(axis=1)
    not_both_sig = np.logical_not(both_sig)
    
    plt.scatter(fc_df.loc[not_both_sig, 'sca_lfc'],
                fc_df.loc[not_both_sig, 'intron_lfc'],
                alpha=0.1, c='grey', label='All genes')
    plt.scatter(fc_df.loc[both_sig, 'sca_lfc'],
                fc_df.loc[both_sig, 'intron_lfc'],
                alpha=0.2, c=sig_color, label='Differential genes')
    
    print({k:v.sum() for k, v in get_concordance(fc_df).items()})

    plt.text(-7, -7, '{:.0%}'.format(
        get_concordance(fc_df)['both_down'].mean()))
    plt.text(7, 8.5, '{:.0%}'.format(
        get_concordance(fc_df)['both_up'].mean()))
    plt.text(7, -5.8, '{:.0%}'.format(
        get_concordance(fc_df)['discordant_sca_up'].mean()))
    plt.text(-7, 8.5, '{:.0%}'.format(
        get_concordance(fc_df)['discordant_sca_down'].mean()))

    both_sig_df = fc_df.loc[both_sig]
    print({k:v.sum() for k, v in get_concordance(both_sig_df).items()})

    plt.text(-6, -7, '({:.0%})'.format(
            get_concordance(both_sig_df)['both_down'].mean()),
            color=sig_color)
    plt.text(8, 8.5, '({:.0%})'.format(
        get_concordance(both_sig_df)['both_up'].mean()),
            color=sig_color)
    plt.text(8, -5.8, '({:.0%})'.format(
        get_concordance(both_sig_df)['discordant_sca_up'].mean()),
            color=sig_color)
    plt.text(-6, 8.5, '({:.0%})'.format(
        get_concordance(both_sig_df)['discordant_sca_down'].mean()),
            color=sig_color)
    
    plt.scatter(interesting.sca_lfc, interesting.intron_lfc, c='', edgecolors='red')
    
    for gene_id, gene_name in genes_of_interest.items():
        plt.text(
            fc_df.loc[gene_id, 'sca_lfc'] + 0.2,
            fc_df.loc[gene_id, 'intron_lfc'],
            gene_name, color='red')

    
    plt.xlim(-8, 9.5)
    plt.ylim(-8, 9.5)
    plt.legend(loc='lower right')
    plt.xlabel('scaRNA log2(fold change)')
    plt.ylabel('intronic RNA log2(fold change)')
